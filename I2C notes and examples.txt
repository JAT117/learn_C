#include <Wire.h>

const int MPU_addr=0x68; //I2C address on the MPU-6050
char AcX_LOW

void setup(){
	Wire.begin();
}

void loop(){
	Wire.beginTransmission(MPU_addr);
	Wire.Write(0x3C); //starting register 0x36(ACCEL_XOUT_H);
	Wire.endTransmission(true);
	AcX_LOW=Wire.read();
}
